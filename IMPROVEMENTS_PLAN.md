    v1_issues = UXIssue.objects.filter(version=v1)
    v2_issues = UXIssue.objects.filter(version=v2)
    
    for issue_v1 in v1_issues:
        # –ò—â–µ–º —Ç—É –∂–µ –ø—Ä–æ–±–ª–µ–º—É –≤ v2 (–ø–æ —Ç–∏–ø—É –∏ URL)
        issue_v2 = v2_issues.filter(
            issue_type=issue_v1.issue_type,
            location_url=issue_v1.location_url
        ).first()
        
        if issue_v2:
            if issue_v2.affected_sessions < issue_v1.affected_sessions * 0.5:
                # –ü—Ä–æ–±–ª–µ–º–∞ —Ä–µ—à–µ–Ω–∞ (—É–º–µ–Ω—å—à–∏–ª–∞—Å—å >50%)
                status = 'RESOLVED'

            # –ü—Ä–æ–±–ª–µ–º–∞ –∏—Å—á–µ–∑–ª–∞
            status = 'RESOLVED'
        
        IssueLifecycle.objects.create(
            issue=issue_v1,
            version_first_seen=v1,
            version_resolved=v2 if status == 'RESOLVED' else None,
            status=status,
            impact_change=issue_v2.impact_score - issue_v1.impact_score if issue_v2 else -issue_v1.impact_score
